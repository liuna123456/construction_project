import VueRouter from 'vue-router'; // 导入路由模块 import Home from './components/Home.vue'; // 导入Home组件 import User from './components/User.vue'; import Product from './components/Product.vue';

export default new VueRouter({ // 定义路由规则对象 routes: [ { path: '/home', component: Home }, { path: '/user/:id', component: User }, { path: '/product/:id', component: Product } ] });
